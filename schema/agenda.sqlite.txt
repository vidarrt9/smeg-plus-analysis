CREATE TABLE [ACP] ([ID] INTEGER PRIMARY KEY CHECK(ID >=0), [ACP_STATUS] INTEGER, [ACP_DATE_TIME_YEAR] INTEGER, [ACP_DATE_TIME_MONTH] INTEGER, [ACP_DATE_TIME_DAY] INTEGER, [ACP_DATE_TIME_HOURS] INTEGER, [ACP_DATE_TIME_MINUTES] INTEGER, [ACP_DATE_TIME_SECONDS] INTEGER, [ACP_DATE_TIME_EXP_YEAR] INTEGER, [ACP_DATE_TIME_EXP_MONTH] INTEGER, [ACP_DATE_TIME_EXP_DAY] INTEGER, [ACP_DATE_TIME_EXP_HOURS] INTEGER, [ACP_DATE_TIME_EXP_MINUTES] INTEGER, [ACP_DATE_TIME_EXP_SECONDS] INTEGER);
CREATE TABLE [NavigationCoordinate] ([ID] INTEGER PRIMARY KEY CHECK(ID>=0), [NAV_LAT_DEGREE] INTEGER, [NAV_LAT_MIN] INTEGER, [NAV_LAT_SEC] INTEGER, [NAV_LAT_HUN] INTEGER, [NAV_LAT_CARD] INTEGER, [NAV_LON_DEGREE] INTEGER, [NAV_LON_MIN] INTEGER, [NAV_LON_SEC] INTEGER, [NAV_LON_HUN] INTEGER, [NAV_LON_CARD] INTEGER);
CREATE TABLE [PostalAddress] ([ID] INTEGER PRIMARY KEY CHECK(ID>=0), [ADDR_STREET_NUM] TEXT, [ADDR_STREET_NAME] TEXT, [ADDR_POST_CODE] TEXT, [ADDR_TOWN] TEXT, [ADDR_COUNTRY] TEXT);
CREATE TABLE [Email] ([ID] INTEGER CHECK(ID>=0), [EMAIL_TYPE] INTEGER, [EMAIL_ID] TEXT, [EMAIL_INDEX] INTEGER, [EMAIL_DEFAULT] INTEGER);
CREATE TABLE [PhoneNumber] ([ID] INTEGER CHECK(ID>=0), [TEL_TYPE] INTEGER, [NUMBER] TEXT, [TEL_INDEX] INTEGER, [TEL_DEFAULT] INTEGER);
CREATE TABLE [ProfilePicture] ([ID] INTEGER CHECK(ID>=0), [PICTURE_FILE_NAME] TEXT, [PICTURE_FORMAT] INTEGER, [HEIGHT] INTEGER, [WIDTH] INTEGER, [PICTURE_DATA] BLOB);
CREATE TABLE [NavigationAddress] ([ID] INTEGER PRIMARY KEY CHECK(ID>=0), [NAV_TYPE] INTEGER NOT NULL DEFAULT 2, [NAV_ADM_CODE] BLOB, [NAV_ROAD_CODE] BLOB, [NAV_CROSSROAD_CODE] BLOB, [NAV_ADDRESS_TYPE] INTEGER, [NAV_CAT_CODE] BLOB, [NAV_FAC_CODE] BLOB, [NAV_FAC_NAME] TEXT, [NAV_UTM_COORD] BLOB, [NAV_CD_QUARTER] INTEGER NOT NULL DEFAULT 0, [NAV_CD_YEAR] INTEGER NOT NULL DEFAULT 0);
CREATE TABLE [BTPhoneBook] 
(
    [ID] INTEGER PRIMARY KEY CHECK
    (
        ID>200
    ),
    [STATUS] INTEGER,
    [REP_ID] INTEGER,
    [ORIGIN] INTEGER,
    [FIRSTNAME] TEXT,
    [SURNAME] TEXT,
    [NOTE] TEXT,
    [RECO_DATA_VALUE] INTEGER
);
CREATE TABLE [InternalPhoneBook] 
(
    [ID] INTEGER PRIMARY KEY CHECK
    (
        ID >= 0 and ID <= 200
    ),
    [STATUS] INTEGER DEFAULT 1,
    [REP_ID] INTEGER DEFAULT 0,
    [ORIGIN] INTEGER DEFAULT 0,
    [FIRSTNAME] TEXT,
    [SURNAME] TEXT,
    [NOTE] TEXT DEFAULT '""',
    [RECO_DATA_VALUE] INTEGER DEFAULT '-1'
);
CREATE TRIGGER [AgendaElementDelete_1] AFTER DELETE ON [InternalPhoneBook] WHEN old.ID>0 BEGIN insert into AgendaElementIdentifier values(old.ID); END;
CREATE TRIGGER [AgendaElementDelete_2] AFTER DELETE ON [InternalPhoneBook] BEGIN delete from NavigationAddress where ID=old.ID;
delete from PhoneNumber where ID=old.ID;
delete from Email where ID=old.ID;
delete from ProfilePicture where ID=old.ID;
delete from NavigationCoordinate where ID=old.ID;
delete from PostalAddress where ID=old.ID;
delete from ACP where ID=old.ID;
 END;
CREATE TRIGGER [AgendaElementInsert] AFTER INSERT ON [InternalPhoneBook] BEGIN delete from AgendaElementIdentifier where ID=new.ID and new.ID>0; END;
CREATE TABLE [AgendaElementIdentifier] 
(
    [ID] INTEGER PRIMARY KEY CHECK
    (
        ID > 0 and ID<= 200
    )
);
